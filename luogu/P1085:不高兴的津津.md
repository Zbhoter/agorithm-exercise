# 解答
阅读理解题。注意 nodeJS 的输入输出控制即可。

```JavaScript
const rd = require('readline');

var rl = rd.createInterface({
    input: process.stdin,
    output: process.stdout
})

var maxLine = 0;
var res = 0;
var maxStudyTime = 0;

rl.on( 'line', (input)=>{
    if( maxLine++ < 7){
        let arr = input.split(' ').map( Number);
        if( arr[0] + arr[1] > 8 && arr[0] + arr[1] > maxStudyTime) {
            maxStudyTime = arr[0] + arr[1];
            res = maxLine;
        }
        if( maxLine === 7){
            console.log( res);
            rl.close();
        }
    }
})
```
