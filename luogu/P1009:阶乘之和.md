# 解答
大数乘法，可以归结为高精度问题。

当然，可以使用 python3.. 看讨论区，有小伙伴先用 python3 求解答案，再打表。

此处，考虑到 javascript 中 number 的存储格式非常不方便处理整数，因此使用 cpp.

```cpp
#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string add( string& a, string& b) {
    // 保证 a 的长度小于 b 的长度
    if( a.length() > b.length()) {
        swap( a,b);
    }
    string res = "";
    int i,j;
    int carry = 0;
    for( i = b.length() - 1, j = a.length() - 1; j > -1; i--, j--){
        int temp = a[j] - '0' + b[i] - '0' + carry;
        if( temp > 9) {
            carry = 1;
            temp -= 10;
        }
        else {
            carry = 0;
        }
        res += to_string(temp);
    }
    for( ; i > -1; i--){
        int temp = b[i] - '0' + carry;
        if( temp > 9) {
            carry = 1;
            temp -= 10;
        }
        else {
            carry = 0;
        }
        res += to_string(temp);
    }
    if( carry) {
        res += "1";
    }
    reverse(res.begin(), res.end());
    return res;
}

string multiply( string& num, string base) {
    string res = "0";
    for( int i = base.length() - 1; i > -1; i--) {
        string tempString = "";
        int carry = 0;
        for( int j = num.length() - 1; j > -1; j--) {
            int temp = (base[i] - '0') * (num[j] - '0') + carry;
            if( temp > 9) {
                carry = temp/10;
                temp %= 10;
            }
            else {
                carry = 0;
            }
            tempString += temp + '0';
        }
        if( carry ) {
            tempString += carry + '0';
        }
        reverse( tempString.begin(), tempString.end());
        for( int j = i; j < base.length() - 1; j++){
            tempString += '0';
        }
        res = add(res, tempString);
    }
    return res;
}

int main() {
    int n;
    cin>>n;
    string res = "1";
    string pre = "1";
    for( int i = 2; i <= n; i++) {
        pre = multiply(pre, to_string(i));
        res = add( res, pre);

    }
    cout<<res;
    return 0;
}
```
