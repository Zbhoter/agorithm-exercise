impl Solution {
    pub fn max_sub_array(nums: Vec<i32>) -> i32 {
        if nums.len() == 0 {
            return 0;
        }
        let mut sum = 0;
        let mut max = nums[0];
        for i in 0..nums.len() {
            if sum < 0 {
                sum = 0;
            }
            sum += nums[i];
            max = i32::max(max, sum);
        }
        max
    }
}
